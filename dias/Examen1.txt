n
+176M

w

@crear vg
vgcreate -s 16M datastore /dev/sdb3
vgdisplay datastore

@crear lv
lvcreate -n database datastore -L 160M
mkfs.xfs /dev/datastore/databse
blkid
@agregar en fstab
mkdir /mnt/database
UUID="uuid-number" /mnt/database xfs defaults 0 0
df -h
mount -a
